# Textarea

`ouiInput` is a directive that allows native `<input>` and `<textarea>` elements to work with `<oui-form-field>`. See: https://github.com/scheduleonce/once-ui/blob/master/ui/src/components/input/README.md

Regular Input field

<Canvas>
  <Story
    name="Regular"
    parameters={{
      docs: {
        source: {
          code: `<oui-form-field>
  <input oui-input type="text" placeholder="Type here" >
</oui-form-field>`,
        },
      },
    }}
    args={{
      rows: 2,
      theme: THEME[0],
      disabled: false,
      placeholder: 'Type here',
    }}
    argTypes={{
      theme: {
        options: THEME,
        control: { type: 'select' },
      },
    }}
  >
    {props => ({
      moduleMetadata: {
        imports: [OuiFormFieldModule, OuiInputModule],
        schemas: [],
        declarations: [],
      },
      template: `<oui-form-field ngClass="{{theme}}"> <textarea [disabled]="disabled" [rows]="rows" (blur)="blured()" (focus)="focused()" oui-input [placeholder]="placeholder"></textarea> </oui-form-field>`,
      props: {
        ...props,
        focused: action('focus'),
        blured: action('blur'),
      },
    })}
  </Story>
</Canvas>
